import pyttsx3

def convert_text_to_speech(text):
    engine = pyttsx3.init()
    engine.setProperty('voice', 'voice_files_list')
    engine.setProperty('rate', 150)
    engine.say(text)
    engine.runAndWait()

text = input("Enter the text to convert to speech: ")
convert_text_to_speech(text)

```python
import os

directory = "C:/Program Files (x86)/Steam/steamapps/common/Umineko/sound/voice/09"
voice_samples = []

for filename in os.listdir(directory):
    if filename.endswith(".txt"):
        file_path = os.path.join(directory, filename)
        with open(file_path, "r") as file:
            content = file.read()
            voice_samples.append(content)

output_filepath = os.path.join(directory, "voice_samples.txt")
with open(output_filepath, "w") as output_file:
    for sample in voice_samples:
        output_file.write(sample + "\n")
```
